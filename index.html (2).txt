<h2>Total Commission: $<span id="totalCommission">0</span></h2>

<script>
function updateCommissions() {
  let total = 0;
  const rows = document.querySelectorAll("#commissionTable tbody tr");
  rows.forEach(row => {
    const sales = parseFloat(row.cells[1].querySelector("input").value) || 0;
    const rate = parseFloat(row.cells[2].querySelector("input").value) || 0;
    const commission = (sales * rate / 100).toFixed(2);
    row.cells[3].textContent = commission;
    total += parseFloat(commission);
  });
  document.getElementById("totalCommission").textContent = total.toFixed(2);
}

document.querySelectorAll("input").forEach(input => {
  input.addEventListener("input", updateCommissions);
});

updateCommissions();
</script>
